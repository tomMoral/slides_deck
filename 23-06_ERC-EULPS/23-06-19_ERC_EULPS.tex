\frame{
    \frametitle{EULPS: Event-based unsupervised learning for Physiological Signals}

    \begin{block}{Goal}
        Develop {\bf unsupervised learning} methods to
        jointly model events in {\bf physiological signals}.
    \end{block}

    \vskip1em
    {\bf Core idea:} Factorize the signals' distribution $\mathbb P(X)$ as the distribution of its events $\mathbb P(\{T_k\}_k)$.
    \vskip1em

    \begin{columns}[T]
        \techterm{Anomaly detection}
        \techterm{Classification/Forecasting}
        \techterm{Causality}
    \end{columns}
    \begin{columns}[T]
        \column{.3\textwidth}

    \end{columns}

\frame{
    \frametitle{WP1: Parametric PP to model Physiological Signals Events}

    \begin{block}{Challenge 1}
        Parametric models for $\mathbb P(\{T_k\}_k)$ adapted to physiological signals.
    \end{block}

    \begin{itemize}
        \item Framework of point processes: model event arrival,
        \item Existing methods focus on network interactions,
        \item Location variability is key for signal.
    \end{itemize}
