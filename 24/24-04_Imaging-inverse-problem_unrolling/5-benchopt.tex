\begin{center}
\begin{tabular}{c}
\begin{lstlisting}[language=bash,linewidth=.85\textwidth]
git clone https://github.com/benchopt/benchmark_bilevel
benchopt run ./benchmark_bilevel
\end{lstlisting}
\end{tabular}
